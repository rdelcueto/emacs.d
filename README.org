* Introduction
This is a literal Org-mode configuration designed to replicate the mnemonic keybindings of Spacemacs while utilizing a highly granular, domain-driven file tree.

Code blocks in this file are tangled into the `lisp/` directory, and loaded sequentially by `init.el`.

- https://robert.kra.hn/posts/rust-emacs-setup/

* Phase 1: Core, & UI

** Early Initialization (=early-init.el=)
The =early-init.el= file is loaded before the GUI initializes and before the package system starts. It is critical for stripping visual bloat and deferring garbage collection to achieve sub-second daemon startups.

#+begin_src emacs-lisp :tangle early-init.el
  ;;; early-init.el --- Early Init File -*- lexical-binding: t; no-byte-compile: t -*-

  ;; Defer garbage collection during startup
  (setq gc-cons-threshold most-positive-fixnum)

  ;; Disable package.el in favor of Elpaca
  (setq package-enable-at-startup nil)

  ;; Native-compilation settings
  (setq native-comp-jit-compilation t
	native-comp-async-report-warnings-errors 'silent
	native-comp-async-jobs-number
	(max 1 (- (string-to-number (string-trim-right (shell-command-to-string "nproc"))) 1)))

  ;; Performance tweaks
  (setq auto-mode-case-fold nil)
  (setq-default bidi-display-reordering 'left-to-right
	        bidi-paragraph-direction 'left-to-right)
  (setq bidi-inhibit-bpa t)
  (setq inhibit-compacting-font-caches t)
  (setq idle-update-delay 1.0)

  ;; PGTK builds timeout optimization
  (when (boundp 'pgtk-wait-for-event-timeout)
    (setq pgtk-wait-for-event-timeout 0.001))

  ;; Strip GUI bloat BEFORE frame creation (prevents flashing)
  (push '(menu-bar-lines . 0) default-frame-alist)
  (push '(tool-bar-lines . 0) default-frame-alist)
  (push '(vertical-scroll-bars) default-frame-alist)

  (setq menu-bar-mode nil
	tool-bar-mode nil
	scroll-bar-mode nil
	pixel-scroll-precision-mode t
	ring-bell-function 'ignore)

  ;; Frame defaults
  (setq frame-resize-pixelwise t
	frame-inhibit-implied-resize t
	frame-title-format '("%b")
	use-dialog-box t
	use-file-dialog nil
	use-short-answers t
	inhibit-splash-screen t
	inhibit-startup-screen t
	inhibit-startup-echo-area-message user-login-name
	inhibit-startup-buffer-menu t)

  ;; Upgrade built-in packages
  (setq package-install-upgrade-built-in t)
#+end_src

** Package Manager & Module Loader (=init.el=)
The =init.el= file is responsible for bootstrapping the asynchronous =elpaca= package manager, setting up the custom variables file, and loading the domain-driven =lisp/= modules.

#+begin_src emacs-lisp :tangle init.el
    ;;; init.el --- Personal configuration file -*- lexical-binding: t; no-byte-compile: t; -*-
  ;; NOTE: init.el is auto-generated from emacs-config.org.

  ;; 1. Setup Custom File (Isolate auto-generated code)
  (setq custom-file (expand-file-name "custom.el" user-emacs-directory))
  (when (file-exists-p custom-file)
    (load custom-file nil t))
  (setq custom-safe-themes t)

  ;; 2. Bootstrap Elpaca
  (defvar elpaca-installer-version 0.12)
  (defvar elpaca-directory (expand-file-name "elpaca/" user-emacs-directory))
  (defvar elpaca-builds-directory (expand-file-name "builds/" elpaca-directory))
  (defvar elpaca-sources-directory (expand-file-name "sources/" elpaca-directory))
  (defvar elpaca-order '(elpaca :repo "https://github.com/progfolio/elpaca.git"
			      :ref nil :depth 1 :inherit ignore
			      :files (:defaults "elpaca-test.el" (:exclude "extensions"))
			      :build (:not elpaca-activate)))
  (let* ((repo  (expand-file-name "elpaca/" elpaca-sources-directory))
	 (build (expand-file-name "elpaca/" elpaca-builds-directory))
	 (order (cdr elpaca-order))
	 (default-directory repo))
    (add-to-list 'load-path (if (file-exists-p build) build repo))
    (unless (file-exists-p repo)
      (make-directory repo t)
      (when (<= emacs-major-version 28) (require 'subr-x))
      (condition-case-unless-debug err
	(if-let* ((buffer (pop-to-buffer-same-window "*elpaca-bootstrap*"))
		  ((zerop (apply #'call-process `("git" nil ,buffer t "clone"
						  ,@(when-let* ((depth (plist-get order :depth)))
						      (list (format "--depth=%d" depth) "--no-single-branch"))
						  ,(plist-get order :repo) ,repo))))
		  ((zerop (call-process "git" nil buffer t "checkout"
					(or (plist-get order :ref) "--"))))
		  (emacs (concat invocation-directory invocation-name))
		  ((zerop (call-process emacs nil buffer nil "-Q" "-L" "." "--batch"
					"--eval" "(byte-recompile-directory \".\" 0 'force)")))
		  ((require 'elpaca))
		  ((elpaca-generate-autoloads "elpaca" repo)))
	    (progn (message "%s" (buffer-string)) (kill-buffer buffer))
	  (error "%s" (with-current-buffer buffer (buffer-string))))
	((error) (warn "%s" err) (delete-directory repo 'recursive))))
    (unless (require 'elpaca-autoloads nil t)
      (require 'elpaca)
      (elpaca-generate-autoloads "elpaca" repo)
      (let ((load-source-file-function nil)) (load "./elpaca-autoloads"))))
  (add-hook 'after-init-hook #'elpaca-process-queues)
  (elpaca `(,@elpaca-order))

  ;; Enable use-package support
  (elpaca elpaca-use-package
    (elpaca-use-package-mode)
    (setq elpaca-use-package-by-default t))
  (setq use-package-always-ensure t)
  (elpaca-wait)

  (use-package general
    :demand t)
  (elpaca-wait)

  ;; 3. Set Load Path and Require Modules
  (add-to-list 'load-path (expand-file-name "lisp" user-emacs-directory))

  ;; Phase 1: Core & UI
  (require 'cfg-core-defaults)
  (require 'cfg-ui-appearance)
  (elpaca-wait)

  ;; Phase 2: Evil & Keybinding Engine
  (require 'cfg-evil-core)
  (require 'cfg-evil-bindings)
  (elpaca-wait)

  ;; Phase 3
  (require 'cfg-nav-minibuffer)
  (require 'cfg-nav-search)
  (require 'cfg-nav-projects)
  (elpaca-wait)

  ;; Phase 4-6: Everything else
  (require 'cfg-ide-treesit)
  (require 'cfg-ide-lsp)
  (require 'cfg-ide-completion)
  (require 'cfg-ide-tags)
  (require 'cfg-lang-python)
  (require 'cfg-lang-rust)
  (require 'cfg-lang-cc)
  (require 'cfg-lang-lisp)
  (require 'cfg-pkm-org)
  (require 'cfg-pkm-notes)
  (require 'cfg-pkm-reading)
  (require 'cfg-tools-git)
  (require 'cfg-tools-shell)
  (require 'cfg-tools-dired)
  (require 'cfg-tools-ai)
#+end_src

** UI Appearance (=lisp/ui-appearance.el=)
Visual identity: themes, fonts, and modeline.

#+begin_src emacs-lisp :tangle lisp/cfg-ui-appearance.el :mkdirp yes
  ;;; cfg-ui-appearance.el --- Themes and Fonts -*- lexical-binding: t -*-
  (provide 'cfg-ui-appearance)

  ;; 1. Fonts Setup
  (defun rdelcueto/set-fonts-safely (mono-font serif-font symbol-font size)
    "Attempt to set fonts, gracefully catching errors if they are not installed."

    ;; Monospaced Fonts (Default, Fixed Pitch, and Line Numbers)
    (condition-case err
	(progn
	  (set-face-attribute 'default nil :font (font-spec :family mono-font :size size :weight 'regular))
	  (set-face-attribute 'fixed-pitch nil :font (font-spec :family mono-font :size size :weight 'regular))

	  (set-face-attribute 'line-number nil :font mono-font)
	  (set-face-attribute 'line-number-current-line nil
			      :font mono-font
			      :background "gray" :foreground "black"))
      (error (display-warning 'fonts (format "Failed to set monospaced font '%s': %s" mono-font (error-message-string err)) :warning)))

    ;; Variable Pitch (Serif)
    (condition-case err
	(set-face-attribute 'variable-pitch nil :font (font-spec :family serif-font :size size :weight 'medium))
      (error (display-warning 'fonts (format "Failed to set variable-pitch font '%s': %s" serif-font (error-message-string err)) :warning)))

    ;; Symbols (Nerd Font)
    (condition-case err
	(progn
	  (set-fontset-font t 'symbol symbol-font)
	  (set-fontset-font t 'unicode symbol-font))
      (error (display-warning 'fonts (format "Failed to set symbol font '%s': %s" symbol-font (error-message-string err)) :warning))))

  (defun rdelcueto/setup-font-erebus ()
    (rdelcueto/set-fonts-safely "Aporetic Sans Mono" "Aporetic Serif" "JuliaMono Nerd Font" 14.0))

  (defun rdelcueto/setup-font-x201 ()
    (rdelcueto/set-fonts-safely "Aporetic Sans Mono" "Aporetic Serif" "JuliaMono Nerd Font" 12.0))

  ;; Apply fonts dynamically depending on the machine hostname
  (when (string= (system-name) "Erebus")
    (add-hook 'after-init-hook 'rdelcueto/setup-font-erebus)
    (add-hook 'server-after-make-frame-hook 'rdelcueto/setup-font-erebus))

  (when (string= (system-name) "x201")
    (add-hook 'after-init-hook 'rdelcueto/setup-font-x201)
    (add-hook 'server-after-make-frame-hook 'rdelcueto/setup-font-x201))

  ;; 2. Themes
  (use-package doom-themes
    :demand t
    :config
    (setq doom-themes-enable-bold t
	    doom-themes-enable-italic t)
    (doom-themes-visual-bell-config)

    (with-eval-after-load 'org
      (doom-themes-org-config))

    (load-theme 'doom-solarized-dark t))

  ;; 3. Aesthetics (Solaire Mode)
  (use-package solaire-mode
    :demand t
    :config
    (solaire-global-mode +1))
#+end_src

* Phase 3: The Snappy Interface (Navigation & Search)

  (defun lsp-booster--advice-json-parse (old-fn &rest args)
    "Try to parse bytecode instead of json."
    (or
     (when (equal (following-char) ?#)
       (let ((bytecode (read (current-buffer))))
	 (when (byte-code-function-p bytecode)
	 (funcall bytecode))))
     (apply old-fn args)))
  (advice-add (if (progn (require 'json)
		       (fboundp 'json-parse-buffer))
		'json-parse-buffer
	      'json-read)
	    :around
	    #'lsp-booster--advice-json-parse)

  (defun lsp-booster--advice-final-command (old-fn cmd &optional test?)
    "Prepend emacs-lsp-booster command to lsp CMD."
    (let ((orig-result (funcall old-fn cmd test?)))
      (if (and (not test?)                             ;; for check lsp-server-present?
	     (not (file-remote-p default-directory)) ;; see lsp-resolve-final-command, it would add extra shell wrapper
	     lsp-use-plists
	     (not (functionp 'json-rpc-connection))  ;; native json-rpc
	     (executable-find "emacs-lsp-booster"))
	(progn
	  (when-let ((command-from-exec-path (executable-find (car orig-result))))  ;; resolve command from exec-path (in case not found in $PATH)
	    (setcar orig-result command-from-exec-path))
	  (message "Using emacs-lsp-booster for %s!" orig-result)
	  (cons "emacs-lsp-booster" orig-result))
	orig-result)))
  (advice-add 'lsp-resolve-final-command :around #'lsp-booster--advice-final-command)
